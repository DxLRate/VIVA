SQUARE WAVE :-


#include<reg51.h>
void delay();
void main()
{ P2=0x00;
while(1)
{delay();
P2=0xFF;
delay();
P2=0x00;
}
}
void delay()
{unsigned int i;
for(i=0;i<=5000;i++);
}



TRIANGULAR WAVE:- 


#include<reg51.h>
void delay();
void main()
{ unsigned int i;
P2=0x00;
while(1)
{
P2=i;
for(i=0;i<=127;i++)
{
delay();
}
for(i=127;i>0;--i)
{P2=i;
delay();
}
i=0;
}
}
void delay()
{unsigned int i;
for(i=0;i<=5000;i++);
}
